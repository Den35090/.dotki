using System;
using System.Net;

class Program
{
    static void Main()
    {
        string hostName = Dns.GetHostName();
        Console.WriteLine("Имя компьютера: " + hostName);

        IPHostEntry host = Dns.GetHostEntry(hostName);

        foreach (var ip in host.AddressList)
        {
            Console.WriteLine("IP: " + ip.ToString());
        }
    }
}
